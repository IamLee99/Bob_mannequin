Controls =

  Right Arm 
    E, D, R, E

  Left Arm 
    Q, A, W, S
    
  Right Leg 
    I, K, O, L
    
  Left Leg
    Y, H, U, J
    
  Body Control 
    Rotation : Z, X
    Movement : Arrow keys
